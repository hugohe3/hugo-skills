# Markdown Conversion — dependencies
#
# Install everything:
#     pip install -r resources/requirements.txt
#
# Most converters degrade gracefully: if an optional package is missing,
# the corresponding format is just unavailable. PyMuPDF + requests are
# the only hard requirements (PDF parsing + MinerU/web HTTP).

# --- Core (always required) ---
PyMuPDF>=1.23.0          # pdf_to_md.py
requests>=2.31.0         # pdf_to_md_mineru.py, web_to_md.py fallback

# --- Document formats (doc_to_md.py native fast path) ---
mammoth                  # .docx
markdownify              # .html / .epub / web pages
beautifulsoup4           # .html / web pages
ebooklib                 # .epub
nbconvert                # .ipynb

# --- Spreadsheet / slide deck ---
openpyxl                 # .xlsx / .xlsm
python-pptx              # .pptx / .pptm / .ppsx / .potx

# --- Web fetching ---
curl_cffi                # TLS impersonation (WeChat, Cloudflare-protected sites)
pillow                   # image post-processing for web_to_md.py

# --- System binary (not pip-installable, optional) ---
# pandoc — required only for .doc / .odt / .rtf / .tex / .rst / .org / .typ

#   Ubuntu:  sudo apt install pandoc
